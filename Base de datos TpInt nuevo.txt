DROP DATABASE BancoParcial;
create schema BancoParcial;
use BancoParcial;
CREATE TABLE Pais (
    IdPais INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE Provincia (
    IdProvincia INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(50) NOT NULL UNIQUE,
    IdPais INT NOT NULL,
CONSTRAINT Fk_IdPais FOREIGN KEY (IdPais) REFERENCES Pais(IdPais),
 CONSTRAINT UQ_Nombre_IdPais UNIQUE (Nombre, IdPais)
);

CREATE TABLE Localidad (
    IdLocalidad INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(50) NOT NULL,
    IdProvincia INT NOT NULL,
    FOREIGN KEY (IdProvincia) REFERENCES Provincia(IdProvincia),
    UNIQUE (Nombre, IdProvincia) -- Evita que se repita la localidad en la misma provincia
);
CREATE TABLE Persona (
    Dni VARCHAR(11) PRIMARY KEY NOT NULL,
    Cuil VARCHAR(22) UNIQUE NOT NULL,
    Nombre VARCHAR(25) NOT NULL,
    Apellido VARCHAR(25) NOT NULL,
    Sexo VARCHAR(10) NOT NULL,
    IdPais INT NOT NULL,
    fechaNacimiento DATE NOT NULL,
    Direccion VARCHAR(50) NOT NULL,
    IdLocalidad INT NOT NULL,
    IdProvincia INT NOT NULL,
    Estado VARCHAR(15) NOT NULL DEFAULT 'Activo',
    CorreoElectronico VARCHAR(50) UNIQUE NOT NULL,
CONSTRAINT Fk_Persona_IdPais FOREIGN KEY (IdPais) REFERENCES Pais(IdPais),
CONSTRAINT Fk_IdProvincia FOREIGN KEY (IdProvincia) REFERENCES Provincia(IdProvincia),
CONSTRAINT Fk_IdLocalidad FOREIGn KEY (IdLocalidad) REFERENCES Localidad(IdLocalidad)
);

CREATE TABLE usuarios (
    IdUsuario INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    NombreUsuario VARCHAR(25) UNIQUE NOT NULL,
    Contraseña VARCHAR(25) NOT NULL,
    dni VARCHAR(11) UNIQUE NOT NULL,
    TipoUsuario BOOLEAN NOT NULL,
    Estado VARCHAR(15) NOT NULL DEFAULT 'Activo',
    CONSTRAINT Fk_Dni_Persona FOREIGN KEY (dni) REFERENCES Persona(Dni)
);

CREATE TABLE TelefonoxPersonas (
    Dni VARCHAR(11) NOT NULL,
    Telefono VARCHAR(11) NOT NULL,
    Estado VARCHAR(15) NOT NULL DEFAULT 'Activo',
    CONSTRAINT Pk_TelefonoxDni PRIMARY KEY (Dni, Telefono),
    CONSTRAINT Fk_Dni FOREIGN KEY (Dni) REFERENCES Persona(Dni)
);

CREATE TABLE TipoCuenta (
    idTipoCuenta INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    descripcion VARCHAR(50) NOT NULL
);

CREATE TABLE Cuentas (
    NroCuenta INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    IdUsuario INT NOT NULL,
    FechaCreacion DATE NOT NULL,
    IdtipoCuenta INT NOT NULL,
    Cbu VARCHAR(22) UNIQUE NOT NULL,
    Saldo DECIMAL(10,2) NOT NULL,
    Estado VARCHAR(15) NOT NULL DEFAULT 'Activa',
    CONSTRAINT Fk_IdUsuario FOREIGN KEY (IdUsuario) REFERENCES usuarios(IdUsuario),
    CONSTRAINT Fk_TipoCuenta FOREIGN KEY (IdtipoCuenta) REFERENCES TipoCuenta(idTipoCuenta)
);

 
CREATE TABLE Prestamos (
    IdPrestamo INT NOT NULL Auto_Increment Primary key,
    IdUsuario INT NOT NULL,
    Fecha DATE NOT NULL,
    ImporteApagar DECIMAL(10,2) NOT NULL,
    ImportePedido DECIMAL(10,2) NOT NULL,
    PlazoDePago VARCHAR(25) NOT NULL,
    MontoCuotasxMes DECIMAL(10,2) NOT NULL,
    EstadoSolicitud VARCHAR(15) NOT NULL DEFAULT 'Pendiente',
    EstadoPago VARCHAR(15) NOT NULL DEFAULT 'En curso',
    IdCuenta INT NOT NULL,
    CONSTRAINT FK_Prestamos_Usuario FOREIGN KEY (IdUsuario) REFERENCES usuarios(IdUsuario),
    CONSTRAINT FK_Prestamos_Cuenta FOREIGN KEY (IdCuenta) REFERENCES Cuentas(NroCuenta)
);
CREATE TABLE TipoMovimiento (
    IdTipoMovimiento INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    Descripcion VARCHAR(50) NOT NULL
);
CREATE TABLE Movimientos (
    IdMovimiento INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    IdUsuario INT NOT NULL,
    IdTipoMovimiento INT NOT NULL,
    CuentaEmisor  INT NULL,
    Detalle VARCHAR(50) NOT NULL,
    Importe DECIMAL(10,2) NOT NULL,
    Fecha DATE NOT NULL,
    CuentaReceptor INT NULL,
    Estado VARCHAR(15) NOT NULL DEFAULT 'Activo',
    CONSTRAINT FK_Movimientos_Usuario FOREIGN KEY (IdUsuario) REFERENCES usuarios(IdUsuario),
    CONSTRAINT FK_Movimientos_TipoMovimiento FOREIGN KEY (IdTipoMovimiento) REFERENCES TipoMovimiento(IdTipoMovimiento),
     CONSTRAINT FK_Movimientos_CuentaReceptor FOREIGN KEY (CuentaReceptor) REFERENCES Cuentas(NroCuenta),
     CONSTRAINT FK_Movimientos_CuentaEmisor FOREIGN KEY (CuentaEmisor) REFERENCES Cuentas(NroCuenta)
);


CREATE TABLE Cuotas (
    IdCuota INT AUTO_INCREMENT PRIMARY KEY,
    NroCuota INT NOT NULL,
    Importe DECIMAL(15,2) NOT NULL,
    FechaVencimiento DATE NOT NULL,
    FechaPago DATE NULL,
    IdPrestamo INT NOT NULL,
    Estado VARCHAR(15) NOT NULL DEFAULT 'Activo',
    FOREIGN KEY (IdPrestamo) REFERENCES Prestamos(IdPrestamo)
);
